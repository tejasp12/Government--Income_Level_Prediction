def rent_payment(cols):
    rent_amount = cols[0]
    no_of_rooms = cols[1]
    
    if pd.isnull(rent_amount):
        if no_of_rooms == 1:
            return 63344.814815
        elif no_of_rooms == 2:
            return 93637.534247
        elif no_of_rooms == 3:
            return 114242.625850
        elif no_of_rooms == 4:
            return 120205.598203
        elif no_of_rooms == 5:
            return 154373.894611
        elif no_of_rooms == 6:
            return 215407.449782
        elif no_of_rooms == 7:
            return 287232.601399
        elif no_of_rooms == 8:
            return 411316.901639
        elif no_of_rooms == 9:
            return 605221.619048
        elif no_of_rooms == 10:
            return 570540.000000
        elif no_of_rooms == 11:
            return 645475.066875
        else:
            np.nan
